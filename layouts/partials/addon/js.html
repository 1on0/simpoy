{{- if site.Params.enablejs -}} {{- $fingerprint := site.Params.fingerprint -}} {{- $development := eq (hugo.Environment) "development" -}} {{- $darkmode := site.Params.addon.darkmode -}} {{- $clipboard := site.Params.addon.copybutton -}} {{- $highlight := site.Params.addon.highlight -}} {{- $darkmodejs := resources.Get "js/darkmode.js" | js.Build "js/src/darkmode.js" | minify -}} {{- $clipboardjs := resources.Get "js/clipboard.js" | js.Build "js/src/clipboard.js" | minify -}} {{- $hljs := slice (resources.Get "js/highlight.min.js") | resources.Concat "js/src/hljs.js" -}} {{- if and $development $darkmode -}} {{- end -}} {{- if and (eq .Kind "page") $development $clipboard -}} {{- end -}} {{- if and (eq .Kind "page") $development $highlight -}} {{- end -}} {{- if and ($darkmode) (not $development) (not $fingerprint) -}} {{- else if $fingerprint -}} {{- $darkmodefp := $darkmodejs | fingerprint "sha512" -}} {{- end -}} {{- if and ($clipboard) (not $development) (not $fingerprint) -}} {{- else if $fingerprint -}} {{- $clipboardfp := $clipboardjs | fingerprint "sha512" -}} {{- end -}} {{- if and ($highlight) (not $development) (not $fingerprint) -}} {{- else if $fingerprint -}} {{- $hljsfp := $hljs | fingerprint "sha512" -}} {{- end -}} {{- end -}}
